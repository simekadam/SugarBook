 90:                 $aggregation[$row[$this->column]] = $row;

134:     {
135:         if ($this->rows !== NULL) {
136:             return;

139:         $referencing = & $this->refTable->referencing[$this->__toString()];
140:         if ($referencing === NULL) {
141:             $limit = $this->limit;
142:             if ($this->limit && count($this->refTable->rows) > 1) {
143:                 $this->limit = NULL;
144:             }
145:             parent::execute();
146:             $this->limit = $limit;
147:             $referencing = array();
148:             $offset = array();
149:             foreach ($this->rows as $key => $row) {
150:                 $ref = & $referencing[$row[$this->column]];
151:                 $skip = & $offset[$row[$this->column]];
152:                 if ($limit === NULL || (count($ref) < $limit && $skip >= $this->offset)) {
153:                     $ref[$key] = $row;
154:                 } else {
155:                     unset($this->rows[$key]);
156:                 }
157:                 $skip++;
158:                 unset($ref, $skip);
159:             }
160:         }
161: 
162:         $this->data = & $referencing[$this->active];
163:         if ($this->data === NULL) {
