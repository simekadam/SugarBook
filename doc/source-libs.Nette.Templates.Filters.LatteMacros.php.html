 218:      * Finishes parsing.

 223:     {
 224:         // blocks closing check
 225:         if (count($this->blocks) === 1) { // auto-close last block
 226:             $s .= $this->macro('/block');

 228:         } elseif ($this->blocks) {
 229:             throw new LatteException("There are unclosed blocks.", 0, $this->filter->line);

 231: 
 232:         // extends support
 233:         if ($this->namedBlocks || $this->extends) {

 238:     return Nette\Templates\LatteMacros::renderSnippets($control, $_l, get_defined_vars());

 242:     ob_end_clean();
 243:     Nette\Templates\LatteMacros::includeTemplate($_l->extends, get_defined_vars(), $template)->render();

 245: ';
 246:         } else {
 247:             $s = '<?php

 252:         }

 255:         if ($this->namedBlocks) {

 285:     /**
 286:      * Process {macro content | modifiers}
 287:      * @param  string
 288:      * @param  string
 289:      * @param  string
 290:      * @return string
 291:      */
 292:     public function macro($macro, $content = '', $modifiers = '')
 293:     {
 294:         if (func_num_args() === 1) {  // {macro val|modifiers}
 295:             list(, $macro, $content, $modifiers) = String::match($macro, '#^(/?[a-z0-9.:]+)?(.*?)(\\|[a-z](?:'.LatteFilter::RE_STRING.'|[^\'"]+)*)?$()#is');
 296:             $content = trim($content);
 297:         }
 298: 
 299:         if ($macro === '') {
 300:             $macro = substr($content, 0, 2);
 301:             if (!isset($this->macros[$macro])) {
 302:                 $macro = substr($content, 0, 1);
 303:                 if (!isset($this->macros[$macro])) {
 304:                     return FALSE;
 305:                 }
 306:             }
 307:             $content = substr($content, strlen($macro));
 308: 
 309:         } elseif (!isset($this->macros[$macro])) {
 310:             return FALSE;
 311:         }
 312: 
 313:         $closing = $macro[0] === '/';
 314:         if ($closing) {
 315:             $node = array_pop($this->nodes);
 316:             if (!$node || "/$node->name" !== $macro || ($content && !String::startsWith("$node->content ", "$content ")) || $modifiers) {
 317:                 $macro .= $content ? ' ' : '';
 318:                 throw new LatteException("Unexpected macro {{$macro}{$content}{$modifiers}}"
 319:                     . ($node ? ", expecting {/$node->name}" . ($content && $node->content ? " or eventually {/$node->name $node->content}" : '') : ''), 0, $this->filter->line);
 320:             }
 321:             $node->content = $node->modifiers = ''; // back compatibility
 322: 
 323:         } else {
 324:             $node = (object) NULL;
 325:             $node->name = $macro;
 326:             $node->content = $content;
 327:             $node->modifiers = $modifiers;
 328:             if (isset($this->macros["/$macro"])) {
 329:                 $this->nodes[] = $node;
 330:             }
 331:         }
 332: 
 333:         $This = $this;
 334:         return String::replace(
 335:             $this->macros[$macro],
 336:             '#%(.*?)%#',
 337:             function ($m) use ($This, $node) {
 338:                 if ($m[1]) {
 339:                     return callback($m[1][0] === ':' ? array($This, substr($m[1], 1)) : $m[1])
 340:                         ->invoke($node->content, $node->modifiers);
 341:                 } else {
 342:                     return $This->formatMacroArgs($node->content);
 343:                 }
 344:             }
 345:         );
 346:     }

 377:     /**
 378:      * Process <tag n:attr> (experimental).
 379:      * @param  string
 380:      * @param  array
 381:      * @param  bool
 382:      * @return string
 383:      */
 384:     public function attrsMacro($code, $attrs, $closing)
 385:     {
 386:         foreach ($attrs as $name => $content) {
 387:             if (substr($name, 0, 5) === 'attr-') {
 388:                 if (!$closing) {
 389:                     $pos = strrpos($code, '>');
 390:                     if ($code[$pos-1] === '/') $pos--;
 391:                     $code = substr_replace($code, str_replace('@@', substr($name, 5), $this->macro("@attr", $content)), $pos, 0);
 392:                 }
 393:                 unset($attrs[$name]);
 394:             }
 395:         }
 396: 
 397:         $left = $right = array();
 398:         foreach ($this->macros as $name => $foo) {
 399:             if ($name[0] === '@') {
 400:                 $name = substr($name, 1);
 401:                 if (isset($attrs[$name])) {
 402:                     if (!$closing) {
 403:                         $pos = strrpos($code, '>');
 404:                         if ($code[$pos-1] === '/') $pos--;
 405:                         $code = substr_replace($code, $this->macro("@$name", $attrs[$name]), $pos, 0);
 406:                     }
 407:                     unset($attrs[$name]);
 408:                 }
 409:             }
 410: 
 411:             if (!isset($this->macros["/$name"])) { // must be pair-macro
 412:                 continue;
 413:             }
 414: 
 415:             $macro = $closing ? "/$name" : $name;
 416:             if (isset($attrs[$name])) {
 417:                 if ($closing) {
 418:                     $right[] = array($macro, '');
 419:                 } else {
 420:                     array_unshift($left, array($macro, $attrs[$name]));
 421:                 }
 422:             }
 423: 
 424:             $innerName = "inner-$name";
 425:             if (isset($attrs[$innerName])) {
 426:                 if ($closing) {
 427:                     $left[] = array($macro, '');
 428:                 } else {
 429:                     array_unshift($right, array($macro, $attrs[$innerName]));
 430:                 }
 431:             }
 432: 
 433:             $tagName = "tag-$name";
 434:             if (isset($attrs[$tagName])) {
 435:                 array_unshift($left, array($name, $attrs[$tagName]));
 436:                 $right[] = array("/$name", '');
 437:             }
 438: 
 439:             unset($attrs[$name], $attrs[$innerName], $attrs[$tagName]);
 440:         }
 441:         if ($attrs) {
 442:             return FALSE;
 443:         }
 444:         $s = '';
 445:         foreach ($left as $item) {
 446:             $s .= $this->macro($item[0], $item[1]);
 447:         }
 448:         $s .= $code;
 449:         foreach ($right as $item) {
 450:             $s .= $this->macro($item[0], $item[1]);
 451:         }
 452:         return $s;
 453:     }

 481:     /**
 482:      * {syntax ...}
 483:      */
 484:     public function macroSyntax($var)
 485:     {
 486:         switch ($var) {
 487:         case '':
 488:         case 'latte':
 489:             $this->filter->setDelimiters('\\{(?![\\s\'"{}])', '\\}'); // {...}
 490:             break;
 491: 
 492:         case 'double':
 493:             $this->filter->setDelimiters('\\{\\{(?![\\s\'"{}])', '\\}\\}'); // {{...}}
 494:             break;
 495: 
 496:         case 'asp':
 497:             $this->filter->setDelimiters('<%\s*', '\s*%>'); /* <%...%> */
 498:             break;
 499: 
 500:         case 'python':
 501:             $this->filter->setDelimiters('\\{[{%]\s*', '\s*[%}]\\}'); // {% ... %} | {{ ... }}
 502:             break;
 503: 
 504:         case 'off':
 505:             $this->filter->setDelimiters('[^\x00-\xFF]', '');
 506:             break;
 507: 
 508:         default:
 509:             throw new LatteException("Unknown syntax '$var'", 0, $this->filter->line);
 510:         }
 511:     }

 515:     /**
 516:      * {include ...}
 517:      */
 518:     public function macroInclude($content, $modifiers, $isDefinition = FALSE)
 519:     {
 520:         $destination = $this->fetchToken($content); // destination [,] [params]
 521:         $params = $this->formatArray($content) . ($content ? ' + ' : '');
 522: 
 523:         if ($destination === NULL) {
 524:             throw new LatteException("Missing destination in {include}", 0, $this->filter->line);
 525: 
 526:         } elseif ($destination[0] === '#') { // include #block
 527:             $destination = ltrim($destination, '#');
 528:             if (!String::match($destination, '#^' . self::RE_IDENTIFIER . '$#')) {
 529:                 throw new LatteException("Included block name must be alphanumeric string, '$destination' given.", 0, $this->filter->line);
 530:             }
 531: 
 532:             $parent = $destination === 'parent';
 533:             if ($destination === 'parent' || $destination === 'this') {
 534:                 $item = end($this->blocks);
 535:                 while ($item && $item[0] !== self::BLOCK_NAMED) $item = prev($this->blocks);
 536:                 if (!$item) {
 537:                     throw new LatteException("Cannot include $destination block outside of any block.", 0, $this->filter->line);
 538:                 }
 539:                 $destination = $item[1];
 540:             }
 541:             $name = var_export($destination, TRUE);
 542:             $params .= $isDefinition ? 'get_defined_vars()' : '$template->getParams()';
 543:             $cmd = isset($this->namedBlocks[$destination]) && !$parent
 544:                 ? "call_user_func(reset(\$_l->blocks[$name]), \$_l, $params)"
 545:                 : 'Nette\Templates\LatteMacros::callBlock' . ($parent ? 'Parent' : '') . "(\$_l, $name, $params)";
 546:             return $modifiers
 547:                 ? "ob_start(); $cmd; echo " . $this->formatModifiers('ob_get_clean()', $modifiers)
 548:                 : $cmd;
 549: 
 550:         } else { // include "file"
 551:             $destination = $this->formatString($destination);
 552:             $cmd = 'Nette\Templates\LatteMacros::includeTemplate(' . $destination . ', ' . $params . '$template->getParams(), $_l->templates[' . var_export($this->uniq, TRUE) . '])';
 553:             return $modifiers
 554:                 ? 'echo ' . $this->formatModifiers($cmd . '->__toString(TRUE)', $modifiers)
 555:                 : $cmd . '->render()';
 556:         }
 557:     }

1033: 
1034:     /**
1035:      * Tokenizer and preparser.
1036:      * @return array
1037:      */
1038:     private function parseMacro($input)
1039:     {
1040:         $this->tokenizer->tokenize($input);
1041:         $this->tokenizer->tokens[] = NULL; // sentinel
1042: 
1043:         $inTernary = $lastSymbol = $prev = NULL;
1044:         $tokens = $arrays = array();
1045:         $n = -1;
1046:         while (++$n < count($this->tokenizer->tokens)) {
1047:             list($token, $name) = $current = $this->tokenizer->tokens[$n];
1048:             $depth = count($arrays);
1049: 
1050:             if ($name === self::T_COMMENT) {
1051:                 continue; // remove comments
1052: 
1053:             } elseif ($name === self::T_WHITESPACE) {
1054:                 $current[2] = $depth;
1055:                 $tokens[] = $current;
1056:                 continue;
1057: 
1058:             } elseif ($name === self::T_SYMBOL && in_array($prev[0], array(',', '(', '[', '=', '=>', ':', '?', NULL), TRUE)) {
1059:                 $lastSymbol = count($tokens); // quoting pre-requirements
1060: 
1061:             } elseif (is_int($lastSymbol) && in_array($token, array(',', ')', ']', '=', '=>', ':', '|', NULL), TRUE)) {
1062:                 $tokens[$lastSymbol][0] = "'" . $tokens[$lastSymbol][0] . "'"; // quote symbols
1063:                 $lastSymbol = NULL;
1064: 
1065:             } else {
1066:                 $lastSymbol = NULL;
1067:             }
1068: 
1069:             if ($token === '?') { // short ternary operators without :
1070:                 $inTernary = $depth;
1071: 
1072:             } elseif ($token === ':') {
1073:                 $inTernary = NULL;
1074: 
1075:             } elseif ($inTernary === $depth && ($token === ',' || $token === ')' || $token === ']' || $token === NULL)) { // close ternary
1076:                 $tokens[] = array(':', NULL, $depth);
1077:                 $tokens[] = array('null', NULL, $depth);
1078:                 $inTernary = NULL;
1079:             }
1080: 
1081:             if ($token === '[') { // simplified array syntax [...]
1082:                 if ($arrays[] = $prev[0] !== ']' && $prev[1] !== self::T_SYMBOL && $prev[1] !== self::T_VARIABLE) {
1083:                     $tokens[] = array('array', NULL, $depth);
1084:                     $current = array('(', NULL);
1085:                 }
1086:             } elseif ($token === ']') {
1087:                 if (array_pop($arrays) === TRUE) {
1088:                     $current = array(')', NULL);
1089:                 }
1090:             } elseif ($token === '(') { // only count
1091:                 $arrays[] = '(';
1092: 
1093:             } elseif ($token === ')') { // only count
1094:                 array_pop($arrays);
1095:             }
1096: 
1097:             if ($current) {
1098:                 $current[2] = $depth;
1099:                 $tokens[] = $prev = $current;
1100:             }
1101:         }
1102:         return $tokens;
1103:     }
