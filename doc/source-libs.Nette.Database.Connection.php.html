195:      * @param  string  filename
196:      * @return int  count of commands
197:      */
198:     public function loadFile()
199:     {
200:         @set_time_limit(0); // intentionally @
201: 
202:         $handle = @fopen($file, 'r'); // intentionally @
203:         if (!$handle) {
204:             throw new FileNotFoundException("Cannot open file '$file'.");
205:         }
206: 
207:         $count = 0;
208:         $sql = '';
209:         while (!feof($handle)) {
210:             $s = fgets($handle);
211:             $sql .= $s;
212:             if (substr(rtrim($s), -1) === ';') {
213:                 parent::exec($sql); // native query without logging
214:                 $sql = '';
215:                 $count++;
216:             }
217:         }
218:         fclose($handle);
219:         return $count;
220:     }

225:      * Returns syntax highlighted SQL command.
226:      * @param  string
227:      * @return string
228:      */
229:     public static function highlightSql($sql)
230:     {
231:         static $keywords1 = 'SELECT|UPDATE|INSERT(?:\s+INTO)?|REPLACE(?:\s+INTO)?|DELETE|FROM|WHERE|HAVING|GROUP\s+BY|ORDER\s+BY|LIMIT|OFFSET|SET|VALUES|LEFT\s+JOIN|INNER\s+JOIN|TRUNCATE';
232:         static $keywords2 = 'ALL|DISTINCT|DISTINCTROW|AS|USING|ON|AND|OR|IN|IS|NOT|NULL|LIKE|TRUE|FALSE';
233: 
234:         // insert new lines
235:         $sql = " $sql ";
236:         $sql = preg_replace("#(?<=[\\s,(])($keywords1)(?=[\\s,)])#i", "\n\$1", $sql);
237: 
238:         // reduce spaces
239:         $sql = preg_replace('#[ \t]{2,}#', " ", $sql);
240: 
241:         $sql = wordwrap($sql, 100);
242:         $sql = preg_replace("#([ \t]*\r?\n){2,}#", "\n", $sql);
243: 
244:         // syntax highlight
245:         $sql = htmlSpecialChars($sql);
246:         $sql = preg_replace_callback("#(/\\*.+?\\*/)|(\\*\\*.+?\\*\\*)|(?<=[\\s,(])($keywords1)(?=[\\s,)])|(?<=[\\s,(=])($keywords2)(?=[\\s,)=])#is", function($matches) {
247:             if (!empty($matches[1])) // comment
248:                 return '<em style="color:gray">' . $matches[1] . '</em>';
249: 
250:             if (!empty($matches[2])) // error
251:                 return '<strong style="color:red">' . $matches[2] . '</strong>';
252: 
253:             if (!empty($matches[3])) // most important keywords
254:                 return '<strong style="color:blue">' . $matches[3] . '</strong>';
255: 
256:             if (!empty($matches[4])) // other keywords
257:                 return '<strong style="color:green">' . $matches[4] . '</strong>';
258:         }, $sql);
259: 
260:         return '<pre class="dump">' . trim($sql) . "</pre>\n";
261:     }
