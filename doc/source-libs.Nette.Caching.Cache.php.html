109:     /**
110:      * Writes item into the cache.
111:      * Dependencies are:
112:      * - Cache::PRIORITY => (int) priority
113:      * - Cache::EXPIRATION => (timestamp) expiration
114:      * - Cache::SLIDING => (bool) use sliding expiration?
115:      * - Cache::TAGS => (array) tags
116:      * - Cache::FILES => (array|string) file names
117:      * - Cache::ITEMS => (array|string) cache items
118:      * - Cache::CONSTS => (array|string) cache items
119:      *
120:      * @param  mixed  key
121:      * @param  mixed  value
122:      * @param  array  dependencies
123:      * @return mixed  value itself
124:      * @throws \InvalidArgumentException
125:      */
126:     public function save($key, $data, array $dp = NULL)
127:     {
128:         $this->key = is_scalar($key) ? (string) $key : serialize($key);
129:         $key = $this->namespace . md5($this->key);
130: 
131:         // convert expire into relative amount of seconds
132:         if (isset($dp[Cache::EXPIRATION])) {
133:             $dp[Cache::EXPIRATION] = Nette\Tools::createDateTime($dp[Cache::EXPIRATION])->format('U') - time();
134:         }
135: 
136:         // convert FILES into CALLBACKS
137:         if (isset($dp[self::FILES])) {
138:             //clearstatcache();
139:             foreach ((array) $dp[self::FILES] as $item) {
140:                 $dp[self::CALLBACKS][] = array(array(__CLASS__, 'checkFile'), $item, @filemtime($item)); // @ - stat may fail
141:             }
142:             unset($dp[self::FILES]);
143:         }
144: 
145:         // add namespaces to items
146:         if (isset($dp[self::ITEMS])) {
147:             $dp[self::ITEMS] = (array) $dp[self::ITEMS];
148:             foreach ($dp[self::ITEMS] as $k => $item) {
149:                 $dp[self::ITEMS][$k] = $this->namespace . md5(is_scalar($item) ? $item : serialize($item));
150:             }
151:         }
152: 
153:         // convert CONSTS into CALLBACKS
154:         if (isset($dp[self::CONSTS])) {
155:             foreach ((array) $dp[self::CONSTS] as $item) {
156:                 $dp[self::CALLBACKS][] = array(array(__CLASS__, 'checkConst'), $item, constant($item));
157:             }
158:             unset($dp[self::CONSTS]);
159:         }
160: 
161:         if ($data instanceof Nette\Callback || $data instanceof \Closure) {
162:             Nette\Tools::enterCriticalSection();
163:             $data = $data->__invoke();
164:             Nette\Tools::leaveCriticalSection();
165:         }
166: 
167:         if (is_object($data)) {
168:             $dp[self::CALLBACKS][] = array(array(__CLASS__, 'checkSerializationVersion'), get_class($data),
169:                 Nette\Reflection\ClassReflection::from($data)->getAnnotation('serializationVersion'));
170:         }
171: 
172:         $this->data = $data;
173:         if ($data === NULL) {
174:             $this->storage->remove($key);
175:         } else {
176:             $this->storage->write($key, $data, (array) $dp);
177:         }
178:         return $data;
179:     }
