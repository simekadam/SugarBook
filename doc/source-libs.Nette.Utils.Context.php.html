 33:     /**
 34:      * Adds the specified service to the service container.
 35:      * @param  string service name
 36:      * @param  mixed  object, class name or factory callback
 37:      * @param  bool   is singleton?
 38:      * @param  array  factory options
 39:      * @return void
 40:      */
 41:     public function addService($name, $service, $singleton = TRUE, array $options = NULL)
 42:     {
 43:         $this->updating();
 44:         if (!is_string($name) || $name === '') {
 45:             throw new \InvalidArgumentException("Service name must be a non-empty string, " . gettype($name) . " given.");
 46:         }
 47: 
 48:         $lower = strtolower($name);
 49:         if (isset($this->registry[$lower])) { // only for instantiated services?
 50:             throw new AmbiguousServiceException("Service named '$name' has already been registered.");
 51:         }
 52: 
 53:         if ($service instanceof self) {
 54:             $this->registry[$lower] = & $service->registry[$lower];
 55:             $this->factories[$lower] = & $service->factories[$lower];
 56: 
 57:         } elseif (is_object($service) && !($service instanceof \Closure || $service instanceof Callback)) {
 58:             if (!$singleton || $options) {
 59:                 throw new \InvalidArgumentException("Service named '$name' is an instantiated object and must therefore be singleton without options.");
 60:             }
 61:             $this->registry[$lower] = $service;
 62: 
 63:         } else {
 64:             if (!$service) {
 65:                 throw new \InvalidArgumentException("Service named '$name' is empty.");
 66:             }
 67:             $this->factories[$lower] = array($service, $singleton, $options);
 68:         }
 69:     }

 90:     /**
 91:      * Gets the service object of the specified type.
 92:      * @param  string service name
 93:      * @param  array  options in case service is not singleton
 94:      * @return mixed
 95:      */
 96:     public function getService($name, array $options = NULL)
 97:     {
 98:         if (!is_string($name) || $name === '') {
 99:             throw new \InvalidArgumentException("Service name must be a non-empty string, " . gettype($name) . " given.");
100:         }
101: 
102:         $lower = strtolower($name);
103: 
104:         if (isset($this->registry[$lower])) { // instantiated singleton
105:             if ($options) {
106:                 throw new \InvalidArgumentException("Service named '$name' is singleton and therefore can not have options.");
107:             }
108:             return $this->registry[$lower];
109: 
110:         } elseif (isset($this->factories[$lower])) {
111:             list($factory, $singleton, $defOptions) = $this->factories[$lower];
112: 
113:             if ($singleton && $options) {
114:                 throw new \InvalidArgumentException("Service named '$name' is singleton and therefore can not have options.");
115: 
116:             } elseif ($defOptions) {
117:                 $options = $options ? $options + $defOptions : $defOptions;
118:             }
119: 
120:             if (is_string($factory) && strpos($factory, ':') === FALSE) { // class name
121:                 if (!class_exists($factory)) {
122:                     throw new AmbiguousServiceException("Cannot instantiate service '$name', class '$factory' not found.");
123:                 }
124:                 $service = new $factory;
125:                 if ($options) {
126:                     if (method_exists($service, 'setOptions')) {
127:                         $service->setOptions($options); // TODO: better!
128:                     } else {
129:                         throw new \InvalidStateException("Unable to set options, method $factory::setOptions() is missing.");
130:                     }
131:                 }
132: 
133:             } else { // factory callback
134:                 $factory = callback($factory);
135:                 if (!$factory->isCallable()) {
136:                     throw new \InvalidStateException("Cannot instantiate service '$name', handler '$factory' is not callable.");
137:                 }
138:                 $service = $factory($options);
139:                 if (!is_object($service)) {
140:                     throw new AmbiguousServiceException("Cannot instantiate service '$name', value returned by '$factory' is not object.");
141:                 }
142:             }
143: 
144:             if ($singleton) {
145:                 $this->registry[$lower] = $service;
146:                 unset($this->factories[$lower]);
147:             }
148:             return $service;
149: 
150:         } else {
151:             throw new \InvalidStateException("Service '$name' not found.");
152:         }
153:     }
