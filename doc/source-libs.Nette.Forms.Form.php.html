214:      * Cross-Site Request Forgery (CSRF) form protection.

217:      * @return void
218:      */
219:     public function addProtection($message = NULL, $timeout = NULL)
220:     {
221:         $session = $this->getSession()->getNamespace('Nette.Forms.Form/CSRF');
222:         $key = "key$timeout";
223:         if (isset($session->$key)) {
224:             $token = $session->$key;

226:             $session->$key = $token = md5(uniqid('', TRUE));

402:     public function fireEvents()
403:     {
404:         if (!$this->isSubmitted()) {
405:             return;
406: 
407:         } elseif ($this->submittedBy instanceof ISubmitterControl) {
408:             if (!$this->submittedBy->getValidationScope() || $this->isValid()) {
409:                 $this->submittedBy->click();
410:                 $this->onSubmit($this);
411:             } else {
412:                 $this->submittedBy->onInvalidClick($this->submittedBy);
413:                 $this->onInvalidSubmit($this);
414:             }
415: 
416:         } elseif ($this->isValid()) {
417:             $this->onSubmit($this);
418: 
419:         } else {

426:     /**
427:      * Internal: receives submitted HTTP data.
428:      * @return array
429:      */
430:     protected function receiveHttpData()
431:     {
432:         $httpRequest = $this->getHttpRequest();
433:         if (strcasecmp($this->getMethod(), $httpRequest->getMethod())) {
434:             return;
435:         }
436: 
437:         if ($httpRequest->isMethod('post')) {
438:             $data = Nette\ArrayTools::mergeTree($httpRequest->getPost(), $httpRequest->getFiles());
439:         } else {
440:             $data = $httpRequest->getQuery();
441:         }
442: 
443:         if ($tracker = $this->getComponent(self::TRACKER_ID, FALSE)) {
444:             if (!isset($data[self::TRACKER_ID]) || $data[self::TRACKER_ID] !== $tracker->getValue()) {
445:                 return;
446:             }
447:         }
448: 
449:         return $data;
450:     }
