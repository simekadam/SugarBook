 71:     /**
 72:      * Read from cache.
 73:      * @param  string key
 74:      * @return mixed|NULL
 75:      */
 76:     public function read($key)
 77:     {
 78:         $key = $this->prefix . $key;
 79:         $meta = $this->memcache->get($key);
 80:         if (!$meta) return NULL;
 81: 
 82:         // meta structure:
 83:         // array(
 84:         //     data => stored data
 85:         //     delta => relative (sliding) expiration
 86:         //     callbacks => array of callbacks (function, args)
 87:         // )
 88: 
 89:         // verify dependencies
 90:         if (!empty($meta[self::META_CALLBACKS]) && !Cache::checkCallbacks($meta[self::META_CALLBACKS])) {
 91:             $this->memcache->delete($key, 0);
 92:             return NULL;
 93:         }
 94: 
 95:         if (!empty($meta[self::META_DELTA])) {
 96:             $this->memcache->replace($key, $meta, 0, $meta[self::META_DELTA] + time());
 97:         }
 98: 
 99:         return $meta[self::META_DATA];
100:     }

104:     /**
105:      * Writes item into the cache.
106:      * @param  string key
107:      * @param  mixed  data
108:      * @param  array  dependencies
109:      * @return void
110:      */
111:     public function write($key, $data, array $dp)
112:     {
113:         if (isset($dp[Cache::ITEMS])) {
114:             throw new \NotSupportedException('Dependent items are not supported by MemcachedStorage.');
115:         }
116: 
117:         $key = $this->prefix . $key;
118:         $meta = array(
119:             self::META_DATA => $data,
120:         );
121: 
122:         $expire = 0;
123:         if (isset($dp[Cache::EXPIRATION])) {
124:             $expire = (int) $dp[Cache::EXPIRATION];
125:             if (!empty($dp[Cache::SLIDING])) {
126:                 $meta[self::META_DELTA] = $expire; // sliding time
127:             }
128:         }
129: 
130:         if (isset($dp[Cache::CALLBACKS])) {
131:             $meta[self::META_CALLBACKS] = $dp[Cache::CALLBACKS];
132:         }
133: 
134:         if (isset($dp[Cache::TAGS]) || isset($dp[Cache::PRIORITY])) {
135:             if (!$this->context) {
136:                 throw new \InvalidStateException('CacheJournal has not been provided.');
137:             }
138:             $this->getJournal()->write($key, $dp);
139:         }
140: 
141:         $this->memcache->set($key, $meta, 0, $expire);
142:     }
