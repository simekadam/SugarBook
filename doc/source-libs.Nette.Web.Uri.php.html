 93:      * @param  string  URL
 94:      * @throws InvalidArgumentException

 96:     public function __construct($uri = NULL)
 97:     {
 98:         if (is_string($uri)) {
 99:             $parts = @parse_url($uri); // @ - is escalated to exception
100:             if ($parts === FALSE) {
101:                 throw new \InvalidArgumentException("Malformed or unsupported URI '$uri'.");
102:             }
103: 
104:             foreach ($parts as $key => $val) {
105:                 $this->$key = $val;
106:             }
107: 
108:             if (!$this->port && isset(self::$defaultPorts[$this->scheme])) {
109:                 $this->port = self::$defaultPorts[$this->scheme];
110:             }
111: 
112:             if ($this->path === '' && ($this->scheme === 'http' || $this->scheme === 'https')) {
113:                 $this->path = '/';
114:             }
115: 
116:         } elseif ($uri instanceof self) {
117:             foreach ($this as $key => $val) {
118:                 $this->$key = $uri->$key;
119:             }
120:         }

418:      * URI comparsion (this object must be in canonical form).
419:      * @param  string
420:      * @return bool
421:      */
422:     public function isEqual($uri)
423:     {
424:         // compare host + path
425:         $part = self::unescape(strtok($uri, '?#'), '%/');
426:         if (strncmp($part, '//', 2) === 0) { // absolute URI without scheme
427:             if ($part !== '//' . $this->getAuthority() . $this->path) return FALSE;
428: 
429:         } elseif (strncmp($part, '/', 1) === 0) { // absolute path
430:             if ($part !== $this->path) return FALSE;
431: 
432:         } else {
433:             if ($part !== $this->scheme . '://' . $this->getAuthority() . $this->path) return FALSE;
434:         }
435: 
436:         // compare query strings
437:         $part = preg_split('#[&;]#', self::unescape(strtr((string) strtok('?#'), '+', ' '), '%&;=+'));
438:         sort($part);
439:         $query = preg_split('#[&;]#', $this->query);
440:         sort($query);
441:         return $part === $query;
442:     }

470:     /**
471:      * Similar to rawurldecode, but preserve reserved chars encoded.
472:      * @param  string to decode
473:      * @param  string reserved characters
474:      * @return string
475:      */
476:     public static function unescape($s, $reserved = '%;/?:@&=+$,')
477:     {
478:         // reserved (@see RFC 2396) = ";" | "/" | "?" | ":" | "@" | "&" | "=" | "+" | "$" | ","
479:         // within a path segment, the characters "/", ";", "=", "?" are reserved
480:         // within a query component, the characters ";", "/", "?", ":", "@", "&", "=", "+", ",", "$" are reserved.
481:         preg_match_all('#(?<=%)[a-f0-9][a-f0-9]#i', $s, $matches, PREG_OFFSET_CAPTURE | PREG_SET_ORDER);
482:         foreach (array_reverse($matches) as $match) {
483:             $ch = chr(hexdec($match[0][0]));
484:             if (strpos($reserved, $ch) === FALSE) {
485:                 $s = substr_replace($s, $ch, $match[0][1] - 1, 3);
486:             }
487:         }
488:         return $s;
489:     }
